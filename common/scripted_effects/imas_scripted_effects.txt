yearly_cal_spawn_potential_talent={
	export_trigger_value_to_variable={
		trigger = num_assigned_jobs  
		parameters={
			job  = imas_talent_scout
		}
		variable = $VARIABLE$
	}
	change_variable = {
		which = job_talent_seeker_total
		value = $VARIABLE$
	}
	clear_variable = $VARIABLE$

}
